7.1 Nature

# Paths to shapefiles
natur_link <- paste(loc_dir,
                    "DK_SHAPE_UTM32-EUREF89/DK_SHAPE_UTM32-EUREF89/FOT/NATUR",
                    sep = "/")
natur_files <- list.files(path = natur_link,
                          pattern = "*.shp",
                          full.names = TRUE)

# Read all files
natur_sf_int <- future_map_dfr(.x = natur_files,
                               .f = f_shp,
                               .options = furrr_options(seed = T)) 

# Clean dataset
natur_sf <- natur_sf_int %>% 
  # Remove z component
  st_zm() %>% 
  # Names to lower case
  clean_names() %>% 
  # get only the feature type
  select(feat_kode, feat_type) %>% 
  # Convert variables to factors
  mutate(feat_kode = factor(feat_kode),
         feat_type = factor(feat_type)) %>% 
  # Translate into English
  mutate(feat_type_en = case_when(feat_type == "Brugsgrænse" ~ "Usage_limit",
                                  feat_type == "Dæmning" ~ "Dam",
                                  feat_type == "Dige" ~ "Dike",
                                  feat_type == "Gartneri" ~ "Gardening",
                                  feat_type == "Hegn" ~ "Fence",
                                  feat_type == "Krat_Bevoksning" ~ "Scrub_vegetation",
                                  feat_type == "Råstofområde" ~ "Excavation",
                                  feat_type == "Sand_Klit" ~ "Sand_dune",
                                  feat_type == "Skov" ~ "Forest",
                                  feat_type == "Skrænt" ~ "Slope",
                                  feat_type == "Træ" ~ "Tree",
                                  feat_type == "Trægruppe" ~ "Tree_group",
                                  feat_type == "Vådområde" ~ "Wetland"))
# Filter Polygons
natur_sf_poly <- natur_sf %>% 
  filter(st_is(., c("POLYGON", "MULTIPOLYGON")))

# Paths to shapefiles
teknik_link <- paste(loc_dir,
                     "DK_SHAPE_UTM32-EUREF89/DK_SHAPE_UTM32-EUREF89/FOT/TEKNIK",
                     sep = "/")
teknik_files <- list.files(path = teknik_link,
                           pattern = "*.shp",
                           full.names = TRUE)

# Read all files
teknik_sf_int <- future_map_dfr(.x = teknik_files,
                                .f = f_shp,
                                .options = furrr_options(seed = T)) 

# Clean
teknik_sf <- teknik_sf_int %>% 
  # set crs
  st_set_crs(st_crs(natur_sf)) %>% 
  # Remove z component
  st_zm() %>% 
  # Names to lower case
  clean_names() %>% 
  # # get only the feature type
  # select(feat_kode, feat_type) %>% 
  # Convert variables to factors/character
  mutate(across(ends_with("id"), as.character),
         across(starts_with("feat"), as.factor)) %>% 
  # Translate into English
  mutate(feat_type_en = case_when(
    feat_type == "Anlæg diverse" ~ "Other",
    feat_type == "Bassin" ~ "Basin",
    feat_type == "Begravelsesområde" ~ "Burial_area",
    feat_type == "Brønddæksel" ~ "Manhole_cover",
    feat_type == "Bygværk" ~ "Construction" ,
    feat_type == "Høfde" ~ "Groyne",
    feat_type == "Højspændingsledning" ~ "High_voltage_line",
    feat_type == "Mast" ~ "Masts",
    feat_type == "Nedløbsrist" ~ "Drain_grate",
    feat_type == "Parkering" ~ "Parking",
    feat_type == "Startbane" ~ "Runway",
    feat_type == "Statue_Sten" ~ "Statue_stone",
    feat_type == "Teknisk areal" ~ "Technical_area",
    feat_type == "Telemast" ~ "Radio_masts",
    feat_type == "Vindmølle" ~ "Windmill"))
# Filter Polygons
teknik_sf_poly <- teknik_sf %>% 
  filter(st_is(., c("POLYGON", "MULTIPOLYGON"))) 
# Parking areas LINESTRING
parking_lines  <- filter(teknik_sf, feat_type_en == "Parking")

# Convert to MULTIPOINT
parking_points <- st_cast(parking_lines, "POINT") 

# Remove lines with only 3 points (for making POLYGONS we need at least 4 points) 
nps <- parking_points %>% 
  group_by(fot_id) %>% 
  summarise(n = n()) %>% 
  ungroup() %>% 
  filter(n < 3)

# b <- a[a$n < 3, ]$fot_id

parking_points <- parking_points %>%  filter(!fot_id %in% nps$fot_id)

# Convert to POLYGONS
parking_polys  <- st_sf(aggregate(x = parking_points$geometry,
                           by = list(parking_points$fot_id),
                           FUN = function(g){st_cast(st_combine(g), "POLYGON")}))
mapview(parking_polys, col.regions = "red") + 
  mapview(parking_lines)

 library(osmdata)

# Area: Copenhagen
  CPH_oms_bb <- getbb("Copenhagen")

# Green areas 
  CPH_parks <- CPH_oms_bb %>%
    opq() %>%
    add_osm_feature(key = "leisure", value = c("park",
                                               "common",
                                               "garden",
                                               "firepit",
                                               "nature_reserve")) %>%
    osmdata_sf() 
  
  CPH_forest <- CPH_oms_bb %>%
    opq() %>%
    add_osm_feature(key = "landuse", value = c("meadow", "allotments", "forest")) %>%
    osmdata_sf() 
  
  CPH_scrub <- CPH_oms_bb %>%
    opq() %>% 
    add_osm_feature(key = "natural", value = "scrub") %>% 
    osmdata_sf()
